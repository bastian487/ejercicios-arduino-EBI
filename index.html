ℹ️
¿Por qué 220Ω?

El LED necesita una resistencia que limite la corriente. Sin ella, el LED se quemaría. 220Ω es el valor estándar para LEDs con Arduino a 5V.

ejercicio_01.ino
// ══════════════════════════════════════════
// Ejercicio 01 — Mi primer LED
// TECuy | EBI-UTU | Asignatura Tecnologías
// ══════════════════════════════════════════

// La constante define el número del pin una sola vez
const int PIN_LED = 13;

void setup() {
  // setup() se ejecuta UNA SOLA VEZ al inicio
  pinMode(PIN_LED, OUTPUT);  // Pin 13 = SALIDA de voltaje
}

void loop() {
  // loop() se repite INFINITAMENTE mientras el Arduino tenga energía
  digitalWrite(PIN_LED, HIGH);  // HIGH = 5V = LED encendido
}

ejercicio_03_boton.ino
// Ejercicio 03 — Botón controla LED
const int PIN_BOTON = 2;
const int PIN_LED   = 13;

void setup() {
  pinMode(PIN_LED,   OUTPUT);
  // INPUT_PULLUP: usa resistencia interna, no necesita externa
  pinMode(PIN_BOTON, INPUT_PULLUP);
}

void loop() {
  int estado = digitalRead(PIN_BOTON);
  
  if (estado == LOW) {
    // LOW = botón presionado (con INPUT_PULLUP)
    digitalWrite(PIN_LED, HIGH);
  } else {
    digitalWrite(PIN_LED, LOW);
  }
}
⚠️
INPUT_PULLUP invierte la lógica

Con INPUT_PULLUP, el pin lee HIGH cuando el botón está libre y LOW cuando está presionado. ¡Es al revés de lo que parece! Esto es normal y muy común en electrónica.

ejercicio_S01_DHT11.ino
// Ejercicio S1 — Sensor DHT11: Temperatura y Humedad
#include <DHT.h>

#define DHT_PIN   2
#define DHT_TYPE  DHT11

DHT dht(DHT_PIN, DHT_TYPE);

void setup() {
  Serial.begin(9600);
  dht.begin();
  Serial.println("Sensor DHT11 iniciado");
  delay(2000); // El DHT11 necesita 2s para estabilizarse
}

void loop() {
  float temperatura = dht.readTemperature();  // en °C
  float humedad     = dht.readHumidity();     // en %
  
  // isnan() detecta si la lectura falló
  if (isnan(temperatura) || isnan(humedad)) {
    Serial.println("ERROR: No se pudo leer el sensor");
    delay(2000);
    return;
  }
  
  Serial.print("Temperatura: ");
  Serial.print(temperatura, 1);
  Serial.print(" °C  |  Humedad: ");
  Serial.print(humedad, 1);
  Serial.println(" %");
  
  delay(2000); // DHT11 requiere mínimo 1s entre lecturas
}

ejercicio_S02_ultrasonido.ino
// Ejercicio S2 — HC-SR04: Sensor Ultrasónico
const int TRIG_PIN = 9;
const int ECHO_PIN = 10;

void setup() {
  Serial.begin(9600);
  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);
}

long medirDistancia() {
  // 1. Enviar pulso de 10µs en TRIG
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);
  
  // 2. Medir tiempo del eco en ECHO
  long duracion = pulseIn(ECHO_PIN, HIGH);
  
  // 3. Calcular distancia: velocidad sonido = 340 m/s = 0.034 cm/µs
  long distancia = duracion * 0.034 / 2; // ÷2 porque es ida y vuelta
  
  return distancia;
}

void loop() {
  long cm = medirDistancia();
  
  if (cm > 2 && cm < 400) { // Rango válido del sensor
    Serial.print("Distancia: ");
    Serial.print(cm);
    Serial.println(" cm");
  } else {
    Serial.println("Fuera de rango");
  }
  delay(500);
}
🔊
¿Cómo funciona el ultrasonido?

El sensor emite un pulso de sonido a 40kHz (inaudible), espera a que rebote en un obstáculo, y mide el tiempo. Con la velocidad del sonido (343 m/s) calcula la distancia. ¡Como el sonar de un delfín!

ejercicio_S04_PIR.ino
// Ejercicio S4 — PIR: Detector con temporizador
const int PIN_PIR = 7;
const int PIN_LED = 13;
const long TIEMPO_ON = 10000; // 10 segundos encendido

bool movimientoDetectado = false;
unsigned long tiempoDeteccion = 0;

void setup() {
  Serial.begin(9600);
  pinMode(PIN_PIR, INPUT);
  pinMode(PIN_LED, OUTPUT);
  Serial.println("Sistema PIR iniciado. Esperando...");
}

void loop() {
  int pir = digitalRead(PIN_PIR);
  
  if (pir == HIGH) {
    // Nuevo movimiento detectado
    movimientoDetectado = true;
    tiempoDeteccion = millis(); // Reinicia el temporizador
    Serial.println("🚨 MOVIMIENTO detectado!");
  }
  
  if (movimientoDetectado) {
    digitalWrite(PIN_LED, HIGH);
    
    // millis() devuelve ms desde que Arduino inició
    if (millis() - tiempoDeteccion >= TIEMPO_ON) {
      movimientoDetectado = false;
      digitalWrite(PIN_LED, LOW);
      Serial.println("✅ Sin movimiento. LED apagado.");
    }
  }
}
⏱️
millis() vs delay() — ¿cuál usar?

Con delay() el Arduino se congela completamente. Con millis() el programa sigue corriendo y podés hacer varias cosas a la vez. Para proyectos reales, siempre preferí millis().

// Ejercicio S5 — Humedad del suelo + Riego automático
const int PIN_SENSOR = A0;
const int PIN_BOMBA  = 13; // Simula bomba de agua con LED
const int SECO       = 600; // Umbral: si > SECO, regar
const int HUMEDO     = 300; // Umbral: si < HUMEDO, parar

void setup() {
  pinMode(PIN_BOMBA, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  int raw = analogRead(PIN_SENSOR);
  
  // map() convierte de 0-1023 a 0-100% (humedad)
  int porciento = map(raw, 1023, 0, 0, 100);
  
  Serial.print("Humedad suelo: ");
  Serial.print(porciento);
  Serial.print("%  (raw: ");
  Serial.print(raw);
  Serial.print(")");
  
  if (raw > SECO) {
    digitalWrite(PIN_BOMBA, HIGH);
    Serial.println(" → 💧 Regando...");
  } else if (raw < HUMEDO) {
    digitalWrite(PIN_BOMBA, LOW);
    Serial.println(" → ✅ Suelo húmedo OK");
  } else {
    Serial.println(" → Nivel normal");
  }
  delay(1000);
}

⚙️ Cómo simular el MQ-2 en Wokwi
  1. Agregá un Potenciómetro desde la biblioteca (busca "Potentiometer")
  2. Conectá el pin central (cursor) del potenciómetro al pin A0
  3. Conectá el extremo izquierdo a GND y el derecho a 5V
  4. Agregá un Buzzer y conectá su pin positivo al PIN 8, negativo a GND
  5. Girando el potenciómetro simulás la concentración de gas
  6. Cuando el valor supera el umbral, se activa la alarma
ejercicio_S06_MQ2.ino
// Ejercicio S6 — MQ-2: Detector de gas con alarma
// En Wokwi: usar potenciómetro en A0 para simular el sensor
const int PIN_MQ2   = A0;
const int PIN_BUZZER = 8;
const int PIN_LED    = 13;
const int UMBRAL_GAS = 600; // 0-1023; ajustar según sensor real

void setup() {
  pinMode(PIN_BUZZER, OUTPUT);
  pinMode(PIN_LED,    OUTPUT);
  Serial.begin(9600);
  Serial.println("Sistema MQ-2 iniciado");
}

void loop() {

// Ejercicio C3 — Puerta automática PIR + Servo
#include <Servo.h>

Servo puerta;
const int PIR_PIN    = 7;
const int SERVO_PIN  = 9;
const int BUZZER_PIN = 8;
const int LED_VERDE  = 13;

// Estados del sistema
enum Estado { CERRADA, ABRIENDO, ABIERTA, CERRANDO };
Estado estadoActual = CERRADA;

unsigned long tiempoAbierta = 0;
const long MAX_ABIERTA = 5000; // 5 segundos
int angulo = 0;

void setup() {
  puerta.attach(SERVO_PIN);
  puerta.write(0);
  pinMode(PIR_PIN,    INPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(LED_VERDE,  OUTPUT);
  Serial.begin(9600);
}

void loop() {
  bool presencia = digitalRead(PIR_PIN);
  
  switch (estadoActual) {
    case CERRADA:
      if (presencia) {
        estadoActual = ABRIENDO;
        tone(BUZZER_PIN, 1000, 200);
        Serial.println("Abriendo...");
      }
      break;
    case ABRIENDO:
      if (angulo < 90) {
        angulo += 5; puerta.write(angulo); delay(30);
      } else {
        estadoActual = ABIERTA;
        tiempoAbierta = millis();
        digitalWrite(LED_VERDE, HIGH);
        Serial.println("Puerta ABIERTA");
      }
      break;
    case ABIERTA:
      if (millis() - tiempoAbierta > MAX_ABIERTA && !presencia) {
        estadoActual = CERRANDO;
        digitalWrite(LED_VERDE, LOW);
        Serial.println("Cerrando...");
      }
      break;
    case CERRANDO:
      if (angulo > 0) {
        angulo -= 5; puerta.write(angulo); delay(30);
      } else {
        estadoActual = CERRADA;
        Serial.println("Puerta CERRADA");
      }
      break;
  }
}

// Ejercicio C4 — LDR + Relay con histéresis
const int PIN_LDR   = A0;
const int PIN_RELAY = 7;
const int UMBRAL_ENCENDER = 400; // Enciende si LDR < 400
const int UMBRAL_APAGAR   = 600; // Apaga si LDR > 600
// Histéresis: banda de 200 puntos evita parpadeo

bool relayEncendido = false;

void setup() {
  // IMPORTANTE: el relay se activa con LOW en muchos módulos
  pinMode(PIN_RELAY, OUTPUT);
  digitalWrite(PIN_RELAY, HIGH); // Inicia apagado
  Serial.begin(9600);
}

void loop() {
  int luz = analogRead(PIN_LDR);
  Serial.print("Luz: "); Serial.print(luz);
  
  if (!relayEncendido && luz < UMBRAL_ENCENDER) {
    relayEncendido = true;
    digitalWrite(PIN_RELAY, LOW); // Activa relay
    Serial.println(" → RELAY ON (oscuro)");
  }
  else if (relayEncendido && luz > UMBRAL_APAGAR) {
    relayEncendido = false;
    digitalWrite(PIN_RELAY, HIGH); // Desactiva relay
    Serial.println(" → RELAY OFF (con luz)");
  } else {
    Serial.println(" → Sin cambio");
  }
  delay(500);
}

// PROYECTO 1 — Sistema de seguridad completo
// Componentes: PIR + LCD I2C + Teclado 4x4 + Buzzer + LEDs
#include <Keypad.h>
#include <LiquidCrystal_I2C.h>

const char CLAVE[] = "1234";
const byte FILAS = 4, COLS = 4;
char teclas[FILAS][COLS] = {
  {'1','2','3','A'},{'4','5','6','B'},
  {'7','8','9','C'},{'*','0','#','D'}
};
byte pF[4]={9,8,7,6}, pC[4]={5,4,3,2};
Keypad kb = Keypad(makeKeymap(teclas), pF, pC, FILAS, COLS);
LiquidCrystal_I2C lcd(0x27,16,2);

const int PIR=A0, LED_R=11, LED_G=12, BUZ=10;
bool armado=false; String entrada="";
int intentos=0;

void lcd2(String l1, String l2) {
  lcd.clear();
  lcd.setCursor(0,0); lcd.print(l1);
  lcd.setCursor(0,1); lcd.print(l2);
}

void setup() {
  lcd.init(); lcd.backlight();
  pinMode(PIR,INPUT); pinMode(LED_R,OUTPUT);
  pinMode(LED_G,OUTPUT); pinMode(BUZ,OUTPUT);
  digitalWrite(LED_G,HIGH); // Verde = sistema OK
  lcd2("Sistema Alarm","# armar  * clave");
}

void loop() {
  char t = kb.getKey();
  
  if (armado && digitalRead(PIR)==HIGH) {
    lcd2("!! INTRUSO !!","Ingrese clave:");
    digitalWrite(LED_R,HIGH);
    tone(BUZ,1000);
  }
  
  if (!t) return;
  
  if (t == '#' && !armado) {
    armado=true;
    lcd2("Sistema ARMADO","Vigilando...");
    digitalWrite(LED_G,LOW); digitalWrite(LED_R,HIGH);
    delay(5000); // Tiempo para salir
    digitalWrite(LED_R,LOW);
    return;
  }
  
  if (t == '*') { entrada=""; lcd2("Clave:",">"); return; }
  
  if (t != '#') {
    entrada += t;
    lcd.setCursor(entrada.length(),1); lcd.print('*');
    return;
  }
  
  if (entrada == CLAVE) {
    armado=false; intentos=0;
    noTone(BUZ); digitalWrite(LED_R,LOW); digitalWrite(LED_G,HIGH);
    lcd2("ACCESO OK!","Sistema off");
    delay(2000);
    lcd2("Sistema Alarm","# armar  * clave");
  } else {
    intentos++;
    lcd2("Clave INCORRECTA", "Intento: "+String(intentos));
    tone(BUZ,300,500);
    if (intentos >= 3) {
      lcd2("BLOQUEADO 30s","Muchos intentos!");
      tone(BUZ,200); delay(30000); noTone(BUZ);
      intentos=0;
    }
  }
  entrada="";
}

// PROYECTO 2 — Medidor de ambiente multi-sensor
// DHT11 + HC-SR04 + LDR + LCD I2C + 2 botones de menú
#include <DHT.h>
#include <LiquidCrystal_I2C.h>

#define DHT_PIN   2
#define DHT_TYPE  DHT11
const int TRIG=9, ECHO=10, LDR=A0;
const int BTN_NEXT=4, BTN_PREV=5;
const int LED_ALERTA=13, BUZ=8;

DHT dht(DHT_PIN, DHT_TYPE);
LiquidCrystal_I2C lcd(0x27,16,2);

int pantalla = 0; // 0=temp, 1=distancia, 2=luz
unsigned long ultimaLectura = 0;
float temp = 0, hum = 0;
long dist = 0;
int  luzVal = 0;

long medirDist() {
  digitalWrite(TRIG,LOW); delayMicroseconds(2);
  digitalWrite(TRIG,HIGH); delayMicroseconds(10);
  digitalWrite(TRIG,LOW);
  return pulseIn(ECHO,HIGH) * 0.034 / 2;
}

void leerSensores() {
  if (millis() - ultimaLectura > 2000) {
    temp = dht.readTemperature();
    hum  = dht.readHumidity();
    dist = medirDist();
    luzVal = analogRead(LDR);
    ultimaLectura = millis();
  }
}

void mostrarPantalla() {
  lcd.clear();
  switch(pantalla) {
    case 0:
      lcd.setCursor(0,0); lcd.print("[1/3] Ambiente");
      lcd.setCursor(0,1);
      lcd.print(temp,1); lcd.print("C  ");
      lcd.print(hum,0); lcd.print("%");
      if (temp > 30) { digitalWrite(LED_ALERTA,HIGH); tone(BUZ,500,200); }
      else digitalWrite(LED_ALERTA,LOW);
      break;
    case 1:
      lcd.setCursor(0,0); lcd.print("[2/3] Distancia");
      lcd.setCursor(0,1); lcd.print(dist); lcd.print(" cm");
      break;
    case 2:
      lcd.setCursor(0,0); lcd.print("[3/3] Luz");
      lcd.setCursor(0,1);
      lcd.print(map(luzVal,0,1023,0,100)); lcd.print("%");
      break;
  }
}

void setup() {
  dht.begin(); lcd.init(); lcd.backlight();
  pinMode(TRIG,OUTPUT); pinMode(ECHO,INPUT);
  pinMode(BTN_NEXT,INPUT_PULLUP); pinMode(BTN_PREV,INPUT_PULLUP);
  pinMode(LED_ALERTA,OUTPUT); pinMode(BUZ,OUTPUT);
  lcd.print("TECuy Medidor");
  lcd.setCursor(0,1); lcd.print("Iniciando...");
  delay(2000);
}

void loop() {
  leerSensores();
  
  if (digitalRead(BTN_NEXT)==LOW) {
    pantalla = (pantalla+1) % 3;
    delay(300);
  }
  if (digitalRead(BTN_PREV)==LOW) {
    pantalla = (pantalla-1+3) % 3;
    delay(300);
  }
  mostrarPantalla();
  delay(500);
}

📋 Sensor S1 — DHT11

Solución completa: El código del ejercicio incluye manejo de errores con isnan().

Errores comunes:
• No agregar el delay(2000) inicial — el DHT11 necesita estabilizarse
• No incluir la resistencia pullup de 10kΩ en la línea DATA
• En Wokwi: olvidar seleccionar "DHT11" y no "DHT22" en el componente

Valores típicos: Temperatura ~20-25°C, Humedad ~40-60% en condiciones normales. El sensor solo funciona en el rango 0-50°C / 20-80%RH.
🚀 Desafío
Agregar una alerta: si la temperatura supera 28°C, que parpadee un LED y suene el buzzer una vez. Evalúa uso de if() con condición comparativa.

📋 Sensor S2 — HC-SR04 Ultrasonido

Explicación de la fórmula: distancia = duracion * 0.034 / 2
La velocidad del sonido es 340 m/s = 0.034 cm/µs. Se divide por 2 porque el pulso hace ida Y vuelta.

Errores comunes:
• Confundir TRIG (salida) con ECHO (entrada)
• No esperar el mínimo de 2µs antes del pulso de TRIG
• Aceptar valores fuera del rango 2-400cm sin filtrar
🚀 Desafío de evaluación
Hacer un termómetro de distancia: que un LED cambie de intensidad (PWM con analogWrite) según la distancia al objeto.

📋 Combinado C1 — DHT11 + LCD

Punto clave: La función mostrarEnLCD() muestra cómo encapsular lógica en funciones propias, uno de los conceptos más importantes de programación estructurada.

Dirección I2C: La dirección 0x27 es la más común, pero algunos módulos usan 0x3F. Si el LCD no muestra nada, probá cambiar esa dirección.

En Wokwi: El LCD I2C aparece como "LCD 1602 I2C". Al hacer clic en él durante la simulación se puede ajustar el contraste.
🚀 Desafío premium
