ℹ️
¿Por qué 220Ω?

El LED necesita limitar la corriente que pasa por él. Sin resistencia, en un proyecto real el LED se quemaría en segundos. 220Ω es el valor estándar para LEDs a 5V.

ejercicio_01.ino
// ══════════════════════════════════════
// Ejercicio 01 — Mi primer LED
// TECuy | EBI-UTU | Tecnologías 2025
// ══════════════════════════════════════

const int PIN_LED = 13; // Definimos el pin una vez

void setup() {
  // setup() corre UNA SOLA VEZ al encender el Arduino
  pinMode(PIN_LED, OUTPUT);  // Pin 13 = SALIDA
}

void loop() {
  // loop() se repite INFINITAMENTE
  digitalWrite(PIN_LED, HIGH);  // HIGH = 5V = LED encendido
}

const int LED_ROJO     = 11;
const int LED_AMARILLO = 12;
const int LED_VERDE    = 13;

void setup() {
  pinMode(LED_ROJO,     OUTPUT);
  pinMode(LED_AMARILLO, OUTPUT);
  pinMode(LED_VERDE,    OUTPUT);
}

void loop() {
  // VERDE — 5 segundos
  digitalWrite(LED_VERDE,    HIGH);
  delay(5000);
  digitalWrite(LED_VERDE,    LOW);

  // AMARILLO — 2 segundos (parpadeo)
  for (int i = 0; i < 4; i++) {
    digitalWrite(LED_AMARILLO, HIGH); delay(250);
    digitalWrite(LED_AMARILLO, LOW);  delay(250);
  }

  // ROJO — 5 segundos
  digitalWrite(LED_ROJO,     HIGH);
  delay(5000);
  digitalWrite(LED_ROJO,     LOW);
}

ejercicio_03_boton.ino
// Ejercicio 03 — Botón controla LED
const int PIN_BOTON = 2;
const int PIN_LED   = 13;

void setup() {
  pinMode(PIN_LED,   OUTPUT);
  pinMode(PIN_BOTON, INPUT_PULLUP); // Resistencia interna
}

void loop() {
  int estado = digitalRead(PIN_BOTON);

  if (estado == LOW) {
    // LOW = botón presionado (lógica invertida con PULLUP)
    digitalWrite(PIN_LED, HIGH);
  } else {
    digitalWrite(PIN_LED, LOW);
  }
}
⚠️
INPUT_PULLUP invierte la lógica

Con INPUT_PULLUP: botón libre = HIGH, botón presionado = LOW. Esto es normal en electrónica y te ahorra una resistencia externa. Con INPUT clásico sería al revés.

S01_DHT11.ino
// Sensor S1 — DHT11: Temperatura y Humedad
#include <DHT.h>

#define DHT_PIN  2
#define DHT_TIPO DHT11

DHT dht(DHT_PIN, DHT_TIPO);

void setup() {
  Serial.begin(9600);
  dht.begin();
  Serial.println("DHT11 iniciado. Esperando 2s...");
  delay(2000); // El DHT11 necesita 2s para estabilizarse
}

void loop() {
  float temp = dht.readTemperature(); // °C
  float hum  = dht.readHumidity();    // %

  if (isnan(temp) || isnan(hum)) {
    Serial.println("Error: no se pudo leer el sensor");
  } else {
    Serial.print("Temp: "); Serial.print(temp, 1); Serial.print(" C  |  ");
    Serial.print("Hum: ");  Serial.print(hum, 1);  Serial.println(" %");
  }
  delay(2000); // DHT11: mínimo 1s entre lecturas
}

S02_ultrasonido.ino
// Sensor S2 — HC-SR04: Distancia por ultrasonido
const int TRIG = 9;
const int ECHO = 10;

void setup() {
  Serial.begin(9600);
  pinMode(TRIG, OUTPUT);
  pinMode(ECHO, INPUT);
}

long medirCm() {
  // 1. Pulso de disparo 10µs
  digitalWrite(TRIG, LOW);  delayMicroseconds(2);
  digitalWrite(TRIG, HIGH); delayMicroseconds(10);
  digitalWrite(TRIG, LOW);
  // 2. Medir tiempo del eco
  long dur = pulseIn(ECHO, HIGH);
  // 3. Distancia = tiempo × 0.034 / 2 (ida y vuelta)
  return dur * 0.034 / 2;
}

void loop() {
  long cm = medirCm();
  if (cm > 2 && cm < 400) {
    Serial.print("Distancia: "); Serial.print(cm); Serial.println(" cm");
  } else {
    Serial.println("Fuera de rango (2-400 cm)");
  }
  delay(500);
}
🔊
¿Cómo funciona?

TRIG envía un pulso de sonido de 40kHz (inaudible). Este rebota en el obstáculo y vuelve a ECHO. Midiendo el tiempo de ida y vuelta y dividiendo entre la velocidad del sonido (340 m/s ≈ 0.034 cm/µs) obtenemos la distancia.

S04_PIR.ino
// Sensor S4 — PIR: Detector con temporizador no bloqueante
const int PIN_PIR  = 7;
const int PIN_LED  = 13;
const long TIEMPO  = 10000; // 10 segundos encendido

bool activo = false;
unsigned long tDeteccion = 0;

void setup() {
  Serial.begin(9600);
  pinMode(PIN_PIR, INPUT);
  pinMode(PIN_LED, OUTPUT);
}

void loop() {
  if (digitalRead(PIN_PIR) == HIGH) {
    activo = true;
    tDeteccion = millis(); // Reinicia temporizador
    Serial.println("Movimiento detectado!");
  }

  if (activo) {
    digitalWrite(PIN_LED, HIGH);
    if (millis() - tDeteccion >= TIEMPO) {
      activo = false;
      digitalWrite(PIN_LED, LOW);
      Serial.println("Sin movimiento. Apagando.");
    }
  }
}
⏱️
millis() vs delay()

delay(10000) congela el Arduino 10 segundos: no puede hacer NADA más. Con millis(), el programa sigue ejecutándose normalmente mientras "espera". Para proyectos reales, siempre usá millis().

⚙️ Cómo simular el MQ-2 en Wokwi
  1. Agregá un Potenciómetro (buscá "Potentiometer") — simulará la concentración de gas
  2. Pin central del potenciómetro → A0 del Arduino
  3. Pin izquierdo → GND, pin derecho → 5V
  4. Agregá un Buzzer — pin (+) → PIN 8, pin (-) → GND
  5. Agregá un LED rojo → PIN 13 vía resistencia 220Ω
  6. Al girar el potenciómetro más de la mitad, se activa la alarma
S05_MQ2.ino
// Sensor S5 — MQ-2: Detector de gas con alarma
// En Wokwi: usá potenciómetro en A0 para simular el sensor
const int PIN_MQ2   = A0;
const int PIN_BUZZER = 8;
const int PIN_LED    = 13;
const int UMBRAL     = 600; // 0-1023

void setup() {
  Serial.begin(9600);
  pinMode(PIN_BUZZER, OUTPUT);
  pinMode(PIN_LED,    OUTPUT);
}

void loop() {
  int gas = analogRead(PIN_MQ2);
  Serial.print("Gas: "); Serial.print(gas);

  if (gas > UMBRAL) {
    Serial.println("  ⚠️ GAS DETECTADO!");
    digitalWrite(PIN_LED, HIGH);
    tone(PIN_BUZZER, 1500, 300); delay(300);
    tone(PIN_BUZZER, 800,  300); delay(300);
  } else {
    Serial.println("  ✅ Aire limpio");
    digitalWrite(PIN_LED, LOW);
    noTone(PIN_BUZZER);
    delay(500);
  }
}

// Combinado C1 — DHT11 + LCD I2C
#include <DHT.h>
#include <LiquidCrystal_I2C.h>

DHT dht(2, DHT11);
LiquidCrystal_I2C lcd(0x27, 16, 2);

void mostrar(float t, float h) {
  lcd.clear();
  lcd.setCursor(0,0); lcd.print("Temp: "); lcd.print(t,1); lcd.print("C");
  lcd.setCursor(0,1); lcd.print("Hum:  "); lcd.print(h,1); lcd.print("%");

void setup() {
  dht.begin();
  lcd.init(); lcd.backlight();
  lcd.setCursor(3,0); lcd.print("TECuy");
  lcd.setCursor(1,1); lcd.print("Meteo Station");
  delay(3000);
}

void loop() {
  float t = dht.readTemperature();
  float h = dht.readHumidity();
  if (!isnan(t) && !isnan(h)) mostrar(t, h);
  else { lcd.clear(); lcd.print("Error sensor!"); }
  delay(2500);
}

// Combinado C2 — Sensor de estacionamiento
#include <LiquidCrystal_I2C.h>
LiquidCrystal_I2C lcd(0x27,16,2);

const int TRIG=9,ECHO=10,LV=4,LA=5,LR=6,BUZ=8;

void setup() {
  lcd.init(); lcd.backlight();
  pinMode(TRIG,OUTPUT); pinMode(ECHO,INPUT);
  for(int p:{LV,LA,LR,BUZ}) pinMode(p,OUTPUT);
}

long dist() {
  digitalWrite(TRIG,LOW); delayMicroseconds(2);
  digitalWrite(TRIG,HIGH);delayMicroseconds(10);
  digitalWrite(TRIG,LOW);
  return pulseIn(ECHO,HIGH)*0.034/2;
}

void loop() {
  long cm=dist();

// Combinado C3 — Puerta automática PIR + Servo
#include <Servo.h>
Servo puerta;

const int PIR_PIN   = 7;
const int SERVO_PIN = 9;
const int LED_PIN   = 13;
enum Estado { CERRADA, ABRIENDO, ABIERTA, CERRANDO };
Estado estadoActual = CERRADA;

unsigned long tAbierta = 0;
int angulo = 0;

void setup() {
  puerta.attach(SERVO_PIN); puerta.write(0);
  pinMode(PIR_PIN,INPUT); pinMode(LED_PIN,OUTPUT);
  Serial.begin(9600);
}

void loop() {
  bool hay = digitalRead(PIR_PIN);
  switch(estadoActual) {
    case CERRADA:
      if(hay){ estadoActual=ABRIENDO; Serial.println("Abriendo"); }
      break;
    case ABRIENDO:
      if(angulo<90){ angulo+=3; puerta.write(angulo); delay(20); }
      else{ estadoActual=ABIERTA; tAbierta=millis(); digitalWrite(LED_PIN,HIGH); }
      break;
    case ABIERTA:
      if(millis()-tAbierta>5000 && !hay) estadoActual=CERRANDO;
      break;
    case CERRANDO:
      if(angulo>0){ angulo-=3; puerta.write(angulo); delay(20); }
      else{ estadoActual=CERRADA; digitalWrite(LED_PIN,LOW); Serial.println("Cerrada"); }
      break;
  }
}

// Combinado C4 — LDR + Relay con histéresis
const int PIN_LDR   = A0;
const int PIN_RELAY = 7;
// Histéresis: banda de 200pts evita parpadeo
const int LIM_ON  = 400; // Enciende si LDR < 400 (oscuro)
const int LIM_OFF = 600; // Apaga si LDR > 600 (con luz)

bool relayON = false;

void setup() {
  pinMode(PIN_RELAY, OUTPUT);
  digitalWrite(PIN_RELAY, HIGH); // Relay apagado (activo LOW)
  Serial.begin(9600);
}

void loop() {
  int luz = analogRead(PIN_LDR);
  Serial.print("Luz: "); Serial.print(luz);

  if (!relayON && luz < LIM_ON) {
    relayON = true;
    digitalWrite(PIN_RELAY, LOW); // Activa
    Serial.println("  → Relay ON (oscuro)");
  } else if (relayON && luz > LIM_OFF) {
    relayON = false;
    digitalWrite(PIN_RELAY, HIGH); // Desactiva
    Serial.println("  → Relay OFF (con luz)");
  } else { Serial.println("  → Sin cambio"); }
  delay(500);
}

// PROYECTO 1 — Sistema de alarma con contraseña
// PIR + Teclado 4x4 + LCD I2C + Buzzer + LEDs
#include <Keypad.h>
#include <LiquidCrystal_I2C.h>

const char CLAVE[] = "1234";
const byte F=4,C=4;
char tk[F][C]={{'1','2','3','A'},{'4','5','6','B'},{'7','8','9','C'},{'*','0','#','D'}};
byte pF[4]={9,8,7,6},pC[4]={5,4,3,2};
Keypad kb=Keypad(makeKeymap(tk),pF,pC,F,C);
LiquidCrystal_I2C lcd(0x27,16,2);

const int PIR=A0,LR=11,LG=12,BUZ=10;
bool armado=false; String ent=""; int intentos=0;

void L(String a,String b){lcd.clear();lcd.setCursor(0,0);lcd.print(a);lcd.setCursor(0,1);lcd.print(b);}

void setup() {
  lcd.init();lcd.backlight();
  pinMode(PIR,INPUT);pinMode(LR,OUTPUT);pinMode(LG,OUTPUT);pinMode(BUZ,OUTPUT);
  digitalWrite(LG,HIGH);
  L("Sistema Alarma","# armar * clave");
}

void loop() {
  if(armado && digitalRead(PIR)==HIGH){
    L("!! INTRUSO !!","Ingrese clave:");
    digitalWrite(LR,HIGH); tone(BUZ,1000);
  }
  char t=kb.getKey(); if(!t)return;

  if(t=='#' && !armado){
    armado=true; L("ARMADO","Salir en 5s...");
    digitalWrite(LG,LOW);digitalWrite(LR,HIGH);
    delay(5000); digitalWrite(LR,LOW); return;
  }
  if(t=='*'){ent="";L("Clave:",">");return;}
  if(t!='#'){ent+=t;lcd.setCursor(ent.length(),1);lcd.print('*');return;}

  if(ent==CLAVE){
    armado=false;intentos=0;
    noTone(BUZ);digitalWrite(LR,LOW);digitalWrite(LG,HIGH);
    L("ACCESO OK!","Sistema OFF");
    delay(2000); L("Sistema Alarma","# armar * clave");
  } else {
    intentos++;
    L("INCORRECTO","Intento: "+String(intentos));
    tone(BUZ,300,500);
    if(intentos>=3){
      L("BLOQUEADO 30s","3 intentos!");
      tone(BUZ,200);delay(30000);noTone(BUZ);intentos=0;
    }
  }
  ent="";
}

Soluciones y guía docente

Esta sección contiene soluciones, explicaciones de errores comunes, rúbricas sugeridas y desafíos de evaluación para cada ejercicio. Recomendamos no compartir la URL de esta sección con estudiantes durante las actividades.

📋 Ejercicio 01 — Mi primer LED

Solución: El código mostrado en el ejercicio es la solución completa.

Errores más comunes:
• Olvidar el punto y coma (;) al final de cada instrucción — error de compilación
• Escribir Setup() o Loop() con mayúscula — son case-sensitive
• Conectar el LED al revés (el cátodo debe ir a GND, no al pin)
• Olvidar la resistencia — en hardware real el LED se quema
🚀 Desafío de evaluación

Pedile al estudiante que modifique el código para que el LED parpadee 3 veces rápido (200ms) y luego espere 2 segundos. Evalúa comprensión de delay() y secuencia.

📋 Ejercicio 02 — Semáforo

Solución: El código del ejercicio incluye el parpadeo del amarillo con for().

Errores típicos:
• No apagar el LED anterior antes de encender el siguiente (quedan todos encendidos)
• Olvidar declarar y configurar los 3 pines en setup()

Pregunta pedagógica: "¿Qué pasaría si usamos un solo delay() al final del loop en vez de tres separados?"
🚀 Desafío

Agregar un cuarto estado "noche" (todos apagados) que se active presionando un botón. Evalúa combinación de entradas y salidas.

📋 Ejercicio 03 — Botón y LED

Punto clave: Con INPUT_PULLUP, botón libre = HIGH, presionado = LOW. Es contraintuitivo pero muy común en electrónica.

Errores comunes:
• Usar INPUT en vez de INPUT_PULLUP sin agregar resistencia externa → pin flotante, lecturas erráticas
• Confundir la condición: poner HIGH cuando el botón está presionado (es LOW con PULLUP)
🚀 Desafío avanzado

Hacer que el botón funcione como toggle: un clic enciende, el siguiente apaga. Requiere detectar el flanco de bajada y usar una variable de estado booleana.

📋 Sensor S1 — DHT11

Valores esperados: Temperatura ~20-25°C, Humedad ~45-60% en ambiente de clase.

Errores comunes:
• No agregar delay(2000) en setup — el sensor necesita estabilizarse
• Olvidar la resistencia pullup de 10kΩ en la línea DATA (en hardware real)
• En Wokwi: seleccionar "DHT22" en vez de "DHT11" en las propiedades
• Usar DHT_TYPE DHT22 en el código cuando el sensor es DHT11
🚀 Desafío

Agregar alert visual y sonora: LED rojo parpadea + buzzer si temperatura > 28°C. Evalúa uso de if() con condición comparativa y control de múltiples salidas.

📋 Sensor S2 — HC-SR04

Fórmula: distancia = duración_µs × 0.034 / 2
Velocidad del sonido = 340 m/s = 0.034 cm/µs. Se divide por 2 porque el pulso hace ida Y vuelta.

Errores comunes:
• Invertir TRIG (salida) y ECHO (entrada) en las conexiones o en el código
• No esperar 2µs en LOW antes del pulso de TRIG
• No filtrar valores fuera del rango 2-400cm (el sensor tiene límites físicos)
🚀 Desafío

Crear un theremin visual: que el LED cambie de intensidad usando analogWrite() según la distancia. Requiere map() para convertir 2-400cm a 0-255.

📋 Combinado C1 — DHT11 + LCD I2C

Punto clave: La función mostrar() encapsula la lógica de display. Esto se llama abstracción — concepto fundamental de programación estructurada.

Dirección I2C: La más común es 0x27, pero algunos módulos usan 0x3F. Si el LCD no muestra nada, probá cambiar la dirección.

En Wokwi: Al agregar "LCD 1602 I2C", las conexiones se hacen automáticamente con Autoconnect (clic derecho → autoconnect). La librería ya está preinstalada.
🚀 Desafío premium

Crear pantalla alternada: mostrar temperatura 3s, luego humedad 3s, usando millis() en vez de delay(). Evalúa manejo de tiempo no bloqueante.
